  <script>
    function login() {
      const u = document.getElementById('user').value;
      const p = document.getElementById('pass').value;
      google.script.run.withSuccessHandler(res => {
        if(res.success) {
          document.getElementById('loginSection').classList.add('hidden');
          document.getElementById('mainSection').classList.remove('hidden');
          loadData();
        } else {
          alert("Login Gagal!");
        }
      }).checkLogin(u, p);
    }

    function kirimLaporan() {
      const nohp = document.getElementById('nohp').value;
      if (nohp.length < 11 || nohp.length > 12) {
        alert("Nomor HP harus 11-12 digit!");
        return;
      }

      const fileInput = document.getElementById('foto');
      const file = fileInput.files[0];
      const formData = {
        nama: document.getElementById('nama').value,
        nohp: nohp,
        lembaga: document.getElementById('lembaga').value,
        kejadian: document.getElementById('kejadian').value,
        penanganan: document.getElementById('penanganan').value,
        pihak: document.getElementById('pihak').value
      };

      if (file) {
        const reader = new FileReader();
        reader.onload = function(e) {
          const fileData = {
            data: e.target.result.split(',')[1],
            type: file.type,
            name: file.name
          };
          google.script.run.withSuccessHandler(msg => {
            alert(msg);
            document.getElementById('reportForm').reset();
            loadData();
          }).submitLaporan(formData, fileData);
        };
        reader.readAsDataURL(file);
      } else {
        google.script.run.submitLaporan(formData, null);
      }
    }

    function loadData() {
      google.script.run.withSuccessHandler(data => {
        let html = "";
        for (let i = 1; i < data.length; i++) {
          html += `<tr>
            <td>${new Date(data[i][0]).toLocaleDateString()}</td>
            <td>${data[i][4]}</td>
            <td class="text-primary"><b>${data[i][8] || 'Belum ada tanggapan'}</b></td>
          </tr>`;
        }
        document.getElementById('dataBody').innerHTML = html;
      }).getLaporanData();
    }
  </script>
